# Sidebar-07: Collapsible Icon Sidebar

Based on shadcn/ui blocks sidebar-07 pattern.

## Features
- Collapses to icon-only view
- Breadcrumb navigation in header
- Toggle trigger for expand/collapse
- Responsive breakpoints
- Group-based navigation
- Tooltip on hover when collapsed

### 1. NavMain Component
```tsx
"use client"
import { ChevronRight, type LucideIcon } from "lucide-react"
import {
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from "@/components/ui/collapsible"
import {
  SidebarGroup,
  SidebarGroupLabel,
  SidebarMenu,
  SidebarMenuButton,
  SidebarMenuItem,
  SidebarMenuSub,
  SidebarMenuSubButton,
  SidebarMenuSubItem,
} from "@/components/ui/sidebar"

export function NavMain({
  items,
}: {
  items: {
    title: string
    url: string
    icon?: LucideIcon
    isActive?: boolean
    items?: {
      title: string
      url: string
    }[]
  }[]
}) {
  return (
    <SidebarGroup>
      <SidebarGroupLabel>Platform</SidebarGroupLabel>
      <SidebarMenu>
        {items.map((item) => (
          <Collapsible
            key={item.title}
            asChild
            defaultOpen={item.isActive}
            className="group/collapsible"
          >
            <SidebarMenuItem>
              <CollapsibleTrigger asChild>
                <SidebarMenuButton tooltip={item.title}>
                  {item.icon && <item.icon />}
                  <span>{item.title}</span>
                  <ChevronRight className="ml-auto transition-transform duration-200 group-data-[state=open]/collapsible:rotate-90" />
                </SidebarMenuButton>
              </CollapsibleTrigger>
              <CollapsibleContent>
                <SidebarMenuSub>
                  {item.items?.map((subItem) => (
                    <SidebarMenuSubItem key={subItem.title}>
                      <SidebarMenuSubButton asChild>
                        <a href={subItem.url}>
                          <span>{subItem.title}</span>
                        </a>
                      </SidebarMenuSubButton>
                    </SidebarMenuSubItem>
                  ))}
                </SidebarMenuSub>
              </CollapsibleContent>
            </SidebarMenuItem>
          </Collapsible>
        ))}
      </SidebarMenu>
    </SidebarGroup>
  )
}
```

## Styling Analysis

### Key CSS Classes and Their Purpose
1. **SidebarMenuButton**
   - `size="lg"` - Larger touch target (height: 48px)
   - `data-[state=open]:bg-sidebar-accent` - Active state background
   - `data-[state=open]:text-sidebar-accent-foreground` - Active state text color

2. **Layout Classes**
   - `flex h-16 shrink-0 items-center gap-2` - Header layout
   - `transition-[width,height] ease-linear` - Smooth transitions
   - `group-has-data-[collapsible=icon]/sidebar-wrapper:h-12` - Collapsed state height

3. **Spacing and Dimensions**
   - Avatar: `h-8 w-8 rounded-lg` - 32px square with rounded corners
   - Team logo: `size-8` - 32px square container
   - Icon sizes: `size-4` for small icons, `size-3.5` for team logos
   - Dropdown width: `min-w-56` - Minimum 224px width

4. **Text Styling**
   - `text-sm leading-tight` - Small text with tight line height
   - `truncate` - Ellipsis overflow
   - `font-medium` - Medium font weight for names
   - `text-xs` - Extra small for secondary text

5. **Interactive States**
   - `hover` states handled by SidebarMenuButton internally
   - `pressed` states with opacity changes
   - `data-[state=open]` for dropdown open states

## Universal Component Conversion

### Key Changes for React Native
1. Replace hover states with press states
2. Use Pressable instead of button elements
3. Implement collapsible state management
4. Use React Native animations for transitions
5. Replace CSS classes with style objects
6. Implement tooltip as Popover for mobile

### State Management
```tsx
const [isCollapsed, setIsCollapsed] = useState(false)
const [openGroups, setOpenGroups] = useState<string[]>([])

const toggleGroup = (groupId: string) => {
  setOpenGroups(prev => 
    prev.includes(groupId) 
      ? prev.filter(id => id !== groupId)
      : [...prev, groupId]
  )
}
```

### Responsive Behavior
- Desktop: Full sidebar with collapse button
- Tablet: Start collapsed, expand on hover/press
- Mobile: Drawer-based navigation
